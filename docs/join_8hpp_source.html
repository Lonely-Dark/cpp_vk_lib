11template <typename T, typename Container, typename Delimiter>
12void join_impl(std::string& output, Container&& container, Delimiter&& delimiter)
13{
14 for (const auto& element : container) {
15 if constexpr (std::is_integral_v<T>) {
16 output += std::to_string(element);
17 } else {
18 output += element;
19 }
20 output += delimiter;
21 }
22 if constexpr (std::is_integral_v<std::decay_t<Delimiter>>) {
23 output.pop_back();
24 } else {
25 output.resize(output.size() - delimiter.size());
26 }
27 output.shrink_to_fit();
28}

30template <typename Container, typename Delimiter>
31std::string join(Container&& container, Delimiter&& delimiter)
32{
33 using T = std::decay_t<decltype(*std::begin(std::declval<Container>()))>;
34 std::string output;
35 if constexpr (
36 !std::is_same_v<std::decay_t<Container>, std::forward_list<T>> &&
37 !std::is_same_v<std::decay_t<Container>, std::list<T>>) {
38 static const size_t average_word_length = 20;
39 output.reserve(container.size() * average_word_length);
40 }
41 join_impl<T>(output, container, delimiter);
42 return output;
43}
