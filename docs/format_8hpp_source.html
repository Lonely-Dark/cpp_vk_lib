9template <typename... Args>
10static std::string format(std::string_view data, Args&&... args)
11{
12 if (data.empty()) {
13 return {};
14 }
15
16 static constexpr size_t average_word_size = 7;
17 std::string formatted;
18 formatted.reserve(data.size() + (average_word_size * sizeof...(args)));
19 [[maybe_unused]] auto pack_one = [](auto&& argument) {
20 if constexpr (std::is_integral_v<std::decay_t<decltype(argument)>>) {
21 return std::to_string(argument);
22 } else {
23 return std::string(argument);
24 }
25 };
26 std::array<std::string, sizeof...(Args)> elements{pack_one(args)...};
27 for (size_t i = 0, curr = 0; i < data.size(); i++) {
28 // If we're have '{}' token, insert parameter at this place.
29 if (data[i] == '{' && data[i + 1] == '}') {
30 formatted += elements[curr++];
31 // Add all characters from source string except '{}' token.
32 } else if (data[i - 1] != '{' || data[i] != '}') {
33 formatted += data[i];
34 }
35 }
36 return formatted;
37}
